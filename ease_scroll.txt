        function isTouchDevice() {
            return (('ontouchstart' in window) ||
                (navigator.maxTouchPoints > 0) ||
                (navigator.msMaxTouchPoints > 0));
        }

        if (!isTouchDevice()) {
            let from = 0, to = 0;

            window.addEventListener("wheel", function (e) {
                e.preventDefault();
                to = from + e.deltaY;
                to = Math.max(0, Math.min(document.documentElement.scrollHeight - window.innerHeight, to));
            }, { passive: false })

            function animate() {
                requestAnimationFrame(animate);

                from += (to - from) * 0.07;
                from = +from.toFixed(4).slice(0, -1);
                window.scrollTo(0, from);
            }
            animate();
        }
